**下载/读取文本**

```
import collections
import re
from d2l import torch as d2l

d2l.DATA_HUB['time_machine'] = (d2l.DATA_URL + 'timemachine.txt',
                                '090b5e7e70c295757f55df93cb0a180b9691891a')

def read_time_machine():  
    with open(d2l.download('time_machine'), 'r') as f:
        lines = f.readlines()
    return [re.sub('[^A-Za-z]+', ' ', line).strip().lower() for line in lines]
    #在方括号里的"^"表示非字母的全部，+表示连续至少1个以上
    #sub把line里符合[^A-Za-z]+的都替换为空格。
    #把非大小写英文字母的字符全都变成空格，去除首位空格，全部变小写
lines = read_time_machine()
print(f'# 文本总行数: {len(lines)}')
print(lines[0])
print(lines[10])

**词元化(tokenize)**

```
# 每个文本序列被拆分成一个词元列表，词元（token）是文本的基本单位。
def tokenize(lines, token='word'):      
    if token == 'word':
        #按单词
        return [line.split() for line in lines]
        #按空格分割句子
    elif token == 'char':
        #按字母
        return [list(line) for line in lines]
        #把字符串变成列表，按单个字符
    else:
        print('错误：未知词元类型：' + token)

tokens = tokenize(lines)
for i in range(11):
    print(tokens[i])
```

**定制类封装所有功能呢**

```
#把字符串映射到从0开始的数字索引上
class Vocab:  
    def __init__(self, tokens=None, min_freq=0, reserved_tokens=None):
        # min_freq最少出现的次数
        if tokens is None:
            tokens = []
        if reserved_tokens is None:
            reserved_tokens = []
        # 按出现频率排序
        counter = count_corpus(tokens)
        self._token_freqs = sorted(counter.items(), key=lambda x: x[1],
                                   reverse=True)
        # 从大到小排列
        self.idx_to_token = ['<unk>'] + reserved_tokens
        # 语料库中不存在或已删除的任何词元
        # 尖括号在NLP中代表一类特殊的token
        self.token_to_idx = {token: idx
                             for idx, token in enumerate(self.idx_to_token)}
        #这行代码好像没啥用
        self.idx_to_token, self.token_to_idx = [], dict()
        for token, freq in self._token_freqs: # a Counter
            if freq < min_freq:
                break
                #从大到小排列，少于频率的剔除
            if token not in self.token_to_idx:
                self.idx_to_token.append(token)
                #把token放入token_to_idx列表，同时列表长度加1
                self.token_to_idx[token] = len(self.idx_to_token) - 1
                #把token与在列表的对应索引传入字典

    def __len__(self):
        return len(self.idx_to_token)
        #使实例可以被len()调用

    def __getitem__(self, tokens):
        if not isinstance(tokens, (list, tuple)):
            return self.token_to_idx.get(tokens, self.unk)
            # dict.get(key, default=None)
            # 把不认识的字符做成unknown
            
        return [self.__getitem__(token) for token in tokens]
        #递归，自己调用自己
    #tokens传入的是字符串的列表/元组
    #取出每个token的字符串
    #再调用自己，字符串满足 not isinstance(tokens, (list, tuple)) == True
    #返回其索引

    def to_tokens(self, indices):
        if not isinstance(indices, (list, tuple)):
            return self.idx_to_token[indices]
        #索引时数字
        return [self.idx_to_token[index] for index in indices]
        #索引是列表、元组

    @property
    #装饰器，把属性变方法
    #self.unk = 0
    #是只读的
    def unk(self):  
        return 0

    @property
    #self.token_freqs = self._token_freqs
    def token_freqs(self):
        return self._token_freqs

def count_corpus(tokens):  
    if len(tokens) == 0 or isinstance(tokens[0], list):
        tokens = [token for line in tokens for token in line]
        #for line in tokens:
        #    for token in line:
        #对双重循环的每一个元素，做token操作放在新的列表tokens里
        #这里就把两层嵌套列表变成了一层嵌套
        #tokens的每一行的字符串列表(line)里的单词字符串(token)
        #返回的是整个文本(tokens)所有的字符串
    return collections.Counter(tokens)
    #利用collections.Counter计算每个出现的频率并做成字典（语料）
```

**再用一个函数集成所有功能**

```
vocab = Vocab(tokens)
print(list(vocab.token_to_idx.items())[:10])

def load_corpus_time_machine(max_tokens=-1):  
    lines = read_time_machine()
    tokens = tokenize(lines, 'char')
    vocab = Vocab(tokens)
    # 因为时光机器数据集中的每个文本行不一定是一个句子或一个段落，
    # 所以将所有文本行展平到一个列表中
    corpus = [vocab[token] for line in tokens for token in line]
    # 返回每一个token的索引
    if max_tokens > 0:
        corpus = corpus[:max_tokens]
    return corpus, vocab

corpus, vocab = load_corpus_time_machine()
